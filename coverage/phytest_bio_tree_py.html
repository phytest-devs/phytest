42 def parse_tip_dates( 

43 self, 

44 *, 

45 patterns=None, 

46 date_format: Optional[str] = None, 

47 decimal_year: bool = False, 

48 ): 

49 patterns = patterns or default_date_patterns() 

50 if isinstance(patterns, str): 

51 patterns = [patterns] 

52 

53 dates = {} 

54 

55 compiled_patterns = [re.compile(pattern_string) for pattern_string in patterns] 

56 for tip in self.find_elements(terminal=True): 

57 for pattern in compiled_patterns: 

58 m = pattern.search(tip.name) 

59 if m: 

60 matched_str = m.group(0) 

61 if re.match(r"^\d+\.?\d*$", matched_str): 

62 date = datetime_from_numeric(float(matched_str)) 

63 else: 

64 date = parse(matched_str, date_format) 

65 

66 dates[tip.name] = date 

67 break 

68 

69 if decimal_year: 

70 dates = {key: numeric_date(value) for key, value in dates.items()} 

71 

72 return dates

236 This provides a way to ignore tips that don't follow a loose clock.

239 allow_negative_rate (bool, optional): Whether or not a negative clock rate is allowed. 

240 For trees with little temporal signal, it can be set to True to achieve essentially mid-point rooting. 

241 Defaults to False.

243 If False, then a new optimal root is sought. Defaults to False. 

244 root_method (str, optional): The method used to reroot the tree if `keep_root` is False. 

245 Valid choices are: 'min_dev', 'least-squares', and 'oldest'.
